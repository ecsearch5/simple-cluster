180 int partition(DataType * data, DataType pivot, size_t N, int (*compare)(const DataType *, const DataType *)) {
181  int i=0, j=N-1;
182  while(i <= j) {
183  while((*compare)(&data[i],&pivot) < 0)
184  i++;
185  while((*compare)(&data[j],&pivot) >= 0)
186  j--;
187  if(i <= j) {
188  swap(data,i,j,N);
189  i++;
190  j--;
191  }
192  }
193 
194  return i;
195 }

203 template<typename DataType>
204 void bbsort(DataType * data, size_t N, int (*compare)(const DataType *, const DataType *)) {
205  if(N < 2) return;
206  bool swapped = true;
207  int i, j = 0;
208  while(swapped) {
209  swapped = false;
210  j++;
211  for(i=0;i<N-j;i++) {
212  if((*compare)(&data[i],&data[i+1]) > 0) {
213  swap(data,i,i+1,N);
214  swapped = true;
215  }
216  }
217  }

233 template<typename DataType>
234 size_t quick_select_k(DataType * data, size_t N, size_t k, int (*compare)(const DataType*, const DataType*)) {
235  if(k >= N) {
236  fprintf(stderr, "Out of bounds!\n");
237  exit(1);
238  }
239 
240  if(N <= 8) {
241  bbsort(data,N,*compare);
242  return k;
243  }
244 
245  // First, choose an appropriate pivot
246  DataType pivot = data[N >> 1];
247  // Choose the pivot as the median of left, right and (left+right)/2 elements
248  if(((*compare)(&pivot,&data[0]) <= 0 && (*compare)(&data[0],&data[N-1]) <= 0)
249  || ((*compare)(&pivot,&data[0]) >= 0 && (*compare)(&data[0],&data[N-1]) >= 0))
250  pivot = data[0];
251  if(((*compare)(&pivot,&data[N-1]) <= 0 && (*compare)(&data[N-1],&data[0]) <= 0)
252  || ((*compare)(&pivot,&data[N-1]) >= 0 && (*compare)(&data[N-1],&data[0]) >= 0))
253  pivot = data[N-1];
254 
255  // Then partition the array into two parts based on the pivot value.
256  // The left part will contains members that are less than pivot,
257  // the right part contains member that are greater than or equal pivot.
258  int p = partition(data,pivot,N,*compare);
259  if(p == 0) {
260  bbsort(data,N,*compare);
261  return k;
262  }
263 
264  if(k < p) return quick_select_k(data,p,k,*compare);
265  else if(k == p) return p;
266  else return quick_select_k(&data[p],N-p,k-p,*compare);
267 }
268 }
