192 template<typename DataType>
193 void kd_insert(
194  KDNode<DataType> *& root,
195  DataType * _data,
196  size_t N,
197  size_t level,
198  size_t _id,
199  bool verbose) {
200  // Pre-check
201  if(N <= 0) return;
202  level %= N;
203  if(root == nullptr) {
204  root = ::new KDNode<DataType>(N);
205  root->add_data(_data);
206  root->id = _id;
207  return;
208  }
209 
210  if(root->id == _id) {
211  if(verbose)
212  cerr << "No new node will be inserted! "
213  "Constraint: Id field is an unique field! "
214  "Conflicted at id:" << _id << endl;
215  return;
216  }
217 
218  if(root->at(level) < _data[level]) {
219  kd_insert(root->right,_data,N,(level+1)%N,_id,verbose);
220  } else if(root->at(level) >= _data[level]) {
221  kd_insert(root->left,_data,N,(level+1)%N,_id,verbose);
222  }
223 }
