142 void kd_insert(KDNode<DataType> *& root, const DataType * _data,
143  size_t N, size_t level, size_t _id, bool verbose) {
144  if(root == NULL) {
145  root = ::new KDNode<DataType>();
146  for(size_t i = 0; i < N; i++) {
147  root->add_data(_data[i]);
148  root->id = _id;
149  }
150  return;
151  }
152 
153  if(root->id == _id) {
154  if(verbose)
155  cerr << "No new node will be inserted! "
156  "Constraint: Id field is an unique field! "
157  "Conflicted at id:" << _id << endl;
158  return;
159  }
160 
161  if(root->get_data_at(level) < _data[level]) {
162  kd_insert(root->right,_data,N,(level+1)%N,_id,verbose);
163  } else if(root->get_data_at(level) >= _data[level]) {
164  kd_insert(root->left,_data,N,(level+1)%N,_id,verbose);
165  }
166 }
